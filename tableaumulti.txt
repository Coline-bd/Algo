Exercice 2 : Notes de plusieurs classes

notes <- [
		[10,12,8,19,17],
		[10,12,8,19,17],
		[10,12,8,19,17]
	]
//calcul moyenne de la 2ème classe
somme <- 0
POUR j de 0 à taille de notes[1]-1
	somme <- somme + notes[1][j]
FIN POUR
AFFICHER somme / taille de notes[1]


Version avec une fonction----------------------

Sous-algorithme moyenneClasse(tableau)
	somme <-0
	POUR i de 0 à taille de tableau -1
		somme <- somme + tableau[i]
	FIN POUR
	moyenne <- somme / taille de tableau
	RETOURNER moyenne
	FIN SOUS-ALGORITHME

Exercice 3 : Chercher dans une grille

nombres <- [
		[6,3,5,4],
		[5,2,3,1],
		[8,5,4,2],
		[2,9,4,6]
	]

SOUS ALGORITHME chercherNombre(tableau, target)
	trouver <- faux
	POUR i de 0 à taille de tableau-1
        	POUR j de 0 à taille de tableau[i]-1
			SI tableau[i][j]==target
				AFFICHER "Valeur trouvée en ligne " + i + "colonne" + j
				trouver <- vrai
			FIN SI
		FIN POUR
	FIN POUR
	SI trouve == faux
		AFFICHER "Nombre non trouvé"
	FIN SI 
	FIN SOUS-ALGORITHME
